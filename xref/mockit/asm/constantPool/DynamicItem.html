3   import javax.annotation.Nonnegative;
4   import javax.annotation.Nonnull;

6   import mockit.asm.jvmConstants.ConstantPoolTypes;

8   public final class DynamicItem extends TypeOrMemberItem {
9       @Nonnegative
10      int bsmIndex;
11  
12      public DynamicItem(@Nonnegative int index) {
13          super(index);
14      }
15  
16      DynamicItem(@Nonnegative int index, @Nonnull DynamicItem item) {
17          super(index, item);
18          bsmIndex = item.bsmIndex;
19      }
20  
21      /**
22       * Sets the type, name, desc, and index of the constant or invoke dynamic item.
23       *
24       * @param type
25       *            one of {@link ConstantPoolTypes#INVOKE_DYNAMIC} or {@link ConstantPoolTypes#DYNAMIC}, for invoke or
26       *            constant dynamic, respectively
27       * @param name
28       *            the item name
29       * @param desc
30       *            the item type descriptor
31       * @param index
32       *            zero based index into the class attribute "BootstrapMethods".
33       */
34      public void set(int type, @Nonnull String name, @Nonnull String desc, @Nonnegative int index) {
35          super.type = type;
36          bsmIndex = index;
37          setValuesAndHashcode(name, desc, index);
38      }
39  
40      @Override
41      boolean isEqualTo(@Nonnull Item item) {
42          DynamicItem other = (DynamicItem) item;
43          return other.bsmIndex == bsmIndex && isEqualTo(other);
44      }
45  }
