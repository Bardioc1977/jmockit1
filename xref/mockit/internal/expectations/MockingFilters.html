7   import java.util.Collection;
8   import java.util.Comparator;
9   import java.util.Enumeration;
10  import java.util.Iterator;
11  import java.util.Map;
12  
13  import javax.annotation.Nonnull;
14  
15  import mockit.asm.types.JavaType;
16  
17  public final class MockingFilters {
18      private MockingFilters() {
19      }
20  
21      public static void validateAsMockable(@Nonnull Class<?> type) {
22          String typeDesc = JavaType.getInternalName(type);
23          validateAsMockable(typeDesc);
24      }
25  
26      public static void validateAsMockable(@Nonnull String typeDesc) {
27          boolean unmockable = ("java/lang/String java/lang/StringBuffer java/lang/StringBuilder java/lang/AbstractStringBuilder "
28                  + "java/util/Iterator java/util/Comparator java/util/Spliterator "
29                  + "java/util/Collection java/util/List java/util/Set java/util/SortedSet java/util/Queue java/util/Enumeration "
30                  + "java/util/Map java/util/SortedMap java/util/Map$Entry java/util/AbstractCollection java/util/AbstractMap "
31                  + "java/util/Hashtable java/lang/Throwable java/lang/Object java/lang/Enum java/lang/System java/lang/ThreadLocal "
32                  + "java/lang/ClassLoader java/lang/Math java/lang/StrictMath java/time/Duration").contains(typeDesc)
33                  || "java/nio/file/Paths".equals(typeDesc) || typeDesc.startsWith("java/util/jar/");
34  
35          if (unmockable) {
36              throw new IllegalArgumentException(typeDesc.replace('/', '.') + " is not mockable");
37          }
38      }
39  
40      public static boolean isSubclassOfUnmockable(@Nonnull Class<?> aClass) {
41          return Collection.class.isAssignableFrom(aClass) || Map.class.isAssignableFrom(aClass)
42                  || Iterator.class.isAssignableFrom(aClass) || Comparator.class.isAssignableFrom(aClass)
43                  || Enumeration.class.isAssignableFrom(aClass) || Throwable.class.isAssignableFrom(aClass)
44                  || ClassLoader.class.isAssignableFrom(aClass) || ThreadLocal.class.isAssignableFrom(aClass)
45                  || Number.class.isAssignableFrom(aClass);
46      }
47  }
