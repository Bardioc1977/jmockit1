5   package mockit.internal.expectations.transformation;
6   
7   import static mockit.asm.jvmConstants.Opcodes.GETFIELD;
8   import static mockit.asm.jvmConstants.Opcodes.SIPUSH;
9   
10  import javax.annotation.Nonnegative;
11  import javax.annotation.Nonnull;
12  
13  import mockit.asm.methods.MethodWriter;
14  import mockit.asm.types.JavaType;
15  
16  final class ArgumentMatching {
17      private static final JavaType[] NO_PARAMETERS = {};
18      private static final String ANY_FIELDS = "any anyString anyInt anyBoolean anyLong anyDouble anyFloat anyChar anyShort anyByte";
19      private static final String WITH_METHODS = "with(Lmockit/Delegate;)Ljava/lang/Object; "
20              + "withAny(Ljava/lang/Object;)Ljava/lang/Object; "
21              + "withCapture()Ljava/lang/Object; withCapture(Ljava/util/List;)Ljava/lang/Object; "
22              + "withCapture(Ljava/lang/Object;)Ljava/util/List; "
23              + "withEqual(Ljava/lang/Object;)Ljava/lang/Object; withEqual(DD)D withEqual(FD)F "
24              + "withInstanceLike(Ljava/lang/Object;)Ljava/lang/Object; "
25              + "withInstanceOf(Ljava/lang/Class;)Ljava/lang/Object; "
26              + "withNotEqual(Ljava/lang/Object;)Ljava/lang/Object; "
27              + "withNull()Ljava/lang/Object; withNotNull()Ljava/lang/Object; "
28              + "withSameInstance(Ljava/lang/Object;)Ljava/lang/Object; "
29              + "withSubstring(Ljava/lang/CharSequence;)Ljava/lang/CharSequence; "
30              + "withPrefix(Ljava/lang/CharSequence;)Ljava/lang/CharSequence; "
31              + "withSuffix(Ljava/lang/CharSequence;)Ljava/lang/CharSequence; "
32              + "withMatch(Ljava/lang/CharSequence;)Ljava/lang/CharSequence;";
33  
34      @Nonnull
35      private final InvocationBlockModifier modifier;
36  
37      // Helper fields that allow argument matchers to be moved to the correct positions of their corresponding
38      // parameters:
39      @Nonnull
40      private final int[] matcherStacks;
41      @Nonnegative
42      private int matcherCount;
43      @Nonnull
44      private JavaType[] parameterTypes;
45  
46      static boolean isAnyField(@Nonnull String name) {
47          return name.startsWith("any") && ANY_FIELDS.contains(name);
48      }
49  
50      static boolean isCallToArgumentMatcher(@Nonnull String name, @Nonnull String desc) {
51          return name.startsWith("with") && WITH_METHODS.contains(name + desc);
52      }
53  
54      ArgumentMatching(@Nonnull InvocationBlockModifier modifier) {
55          this.modifier = modifier;
56          matcherStacks = new int[40];
57          parameterTypes = NO_PARAMETERS;
58      }
59  
60      void addMatcher(@Nonnegative int stackSize) {
61          matcherStacks[matcherCount++] = stackSize;
62      }
63  
64      @Nonnegative
65      int getMatcherCount() {
66          return matcherCount;
67      }
68  
69      @Nonnull
70      JavaType getParameterType(@Nonnegative int parameterIndex) {
71          return parameterTypes[parameterIndex];
72      }
73  
74      void generateCodeToAddArgumentMatcherForAnyField(@Nonnull String fieldOwner, @Nonnull String name,
75              @Nonnull String desc) {
76          MethodWriter mw = modifier.getMethodWriter();
77          mw.visitFieldInsn(GETFIELD, fieldOwner, name, desc);
78          modifier.generateCallToActiveInvocationsMethod(name);
79      }
80  
81      boolean handleInvocationParameters(@Nonnegative int stackSize, @Nonnull String desc) {
82          parameterTypes = JavaType.getArgumentTypes(desc);
83          int stackAfter = stackSize - getSumOfParameterSizes();
84          boolean mockedInvocationUsingTheMatchers = stackAfter < matcherStacks[0];
85  
86          if (mockedInvocationUsingTheMatchers) {
87              generateCallsToMoveArgMatchers(stackAfter);
88              modifier.argumentCapturing.generateCallsToSetArgumentTypesToCaptureIfAny();
89              matcherCount = 0;
90          }
91  
92          return mockedInvocationUsingTheMatchers;
93      }
94  
95      @Nonnegative
96      private int getSumOfParameterSizes() {
97          @Nonnegative
98          int sum = 0;
99  
100         for (JavaType argType : parameterTypes) {
101             sum += argType.getSize();
102         }
103 
104         return sum;
105     }
106 
107     private void generateCallsToMoveArgMatchers(@Nonnegative int initialStack) {
108         @Nonnegative
109         int stack = initialStack;
110         @Nonnegative
111         int nextMatcher = 0;
112         @Nonnegative
113         int matcherStack = matcherStacks[0];
114 
115         for (int i = 0; i < parameterTypes.length && nextMatcher < matcherCount; i++) {
116             stack += parameterTypes[i].getSize();
117 
118             if (stack == matcherStack || stack == matcherStack + 1) {
119                 if (nextMatcher < i) {
120                     generateCallToMoveArgMatcher(nextMatcher, i);
121                     modifier.argumentCapturing.updateCaptureIfAny(nextMatcher, i);
122                 }
123 
124                 nextMatcher++;
125                 matcherStack = matcherStacks[nextMatcher];
126             }
127         }
128     }
129 
130     private void generateCallToMoveArgMatcher(@Nonnegative int originalMatcherIndex, @Nonnegative int toIndex) {
131         MethodWriter mw = modifier.getMethodWriter();
132         mw.visitIntInsn(SIPUSH, originalMatcherIndex);
133         mw.visitIntInsn(SIPUSH, toIndex);
134         modifier.generateCallToActiveInvocationsMethod("moveArgMatcher", "(II)V");
135     }
136 }
