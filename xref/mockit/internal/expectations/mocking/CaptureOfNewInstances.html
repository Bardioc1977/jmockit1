5   package mockit.internal.expectations.mocking;
6   
7   import java.lang.instrument.ClassDefinition;
8   import java.util.ArrayList;
9   import java.util.List;
10  
11  import javax.annotation.Nonnull;
12  import javax.annotation.Nullable;
13  
14  import mockit.asm.classes.ClassReader;
15  import mockit.asm.types.JavaType;
16  import mockit.internal.BaseClassModifier;
17  import mockit.internal.capturing.CaptureOfImplementations;
18  import mockit.internal.startup.Startup;
19  import mockit.internal.state.MockFixture;
20  import mockit.internal.state.TestRun;
21  
22  public final class CaptureOfNewInstances extends CaptureOfImplementations<MockedType> {
23      @Nonnull
24      private final List<Class<?>> baseTypes;
25  
26      CaptureOfNewInstances() {
27          baseTypes = new ArrayList<>();
28      }
29  
30      @Nonnull
31      @Override
32      protected BaseClassModifier createModifier(@Nullable ClassLoader cl, @Nonnull ClassReader cr,
33              @Nonnull Class<?> baseType, @Nullable MockedType typeMetadata) {
34          MockedClassModifier modifier = new MockedClassModifier(cl, cr, typeMetadata);
35          String baseTypeDesc = JavaType.getInternalName(baseType);
36          modifier.setClassNameForCapturedInstanceMethods(baseTypeDesc);
37          return modifier;
38      }
39  
40      @Override
41      protected void redefineClass(@Nonnull Class<?> realClass, @Nonnull byte[] modifiedClass) {
42          ClassDefinition newClassDefinition = new ClassDefinition(realClass, modifiedClass);
43          Startup.redefineMethods(newClassDefinition);
44  
45          MockFixture mockFixture = TestRun.mockFixture();
46          mockFixture.addRedefinedClass(newClassDefinition);
47          mockFixture.registerMockedClass(realClass);
48      }
49  
50      void registerCaptureOfNewInstances(@Nonnull MockedType typeMetadata) {
51          Class<?> baseType = typeMetadata.getClassType();
52  
53          if (!typeMetadata.isFinalFieldOrParameter()) {
54              makeSureAllSubtypesAreModified(typeMetadata);
55          }
56  
57          if (!baseTypes.contains(baseType)) {
58              baseTypes.add(baseType);
59          }
60      }
61  
62      void makeSureAllSubtypesAreModified(@Nonnull MockedType typeMetadata) {
63          Class<?> baseType = typeMetadata.getClassType();
64          makeSureAllSubtypesAreModified(baseType, typeMetadata.fieldFromTestClass, typeMetadata);
65      }
66  
67      public boolean captureNewInstance(@Nonnull Object mock) {
68          Class<?> mockedClass = mock.getClass();
69          return !baseTypes.contains(mockedClass) && isWithCapturing(mockedClass);
70      }
71  
72      private boolean isWithCapturing(@Nonnull Class<?> mockedClass) {
73          Class<?>[] interfaces = mockedClass.getInterfaces();
74  
75          for (Class<?> anInterface : interfaces) {
76              if (baseTypes.contains(anInterface)) {
77                  return true;
78              }
79          }
80  
81          Class<?> superclass = mockedClass.getSuperclass();
82          return superclass != Object.class && (baseTypes.contains(superclass) || isWithCapturing(superclass));
83      }
84  
85      void cleanUp() {
86          baseTypes.clear();
87      }
88  }
