5   package mockit.internal.faking;

7   import static mockit.internal.util.Utilities.getClassType;
8   
9   import java.lang.instrument.ClassDefinition;
10  import java.lang.reflect.Proxy;
11  import java.lang.reflect.Type;
12  
13  import javax.annotation.Nonnull;
14  import javax.annotation.Nullable;
15  
16  import mockit.MockUp;
17  import mockit.asm.classes.ClassReader;
18  import mockit.internal.BaseClassModifier;
19  import mockit.internal.ClassFile;
20  import mockit.internal.startup.Startup;
21  import mockit.internal.state.CachedClassfiles;
22  import mockit.internal.state.TestRun;
23  
24  public final class FakeClassSetup {
25      @Nonnull
26      final Class<?> realClass;
27      @Nullable
28      private ClassReader rcReader;
29      @Nonnull
30      private final FakeMethods fakeMethods;
31      @Nonnull
32      private final MockUp<?> fake;
33      private final boolean forStartupFake;
34  
35      FakeClassSetup(@Nonnull Type fakedType, @Nonnull MockUp<?> fake) {
36          this(getClassType(fakedType), fake, fakedType);
37      }
38  
39      public FakeClassSetup(@Nonnull Class<?> classToFake, @Nonnull MockUp<?> fake, @Nonnull Type fakedType) {
40          realClass = classToFake;
41          this.fake = fake;
42          forStartupFake = Startup.initializing;
43          fakeMethods = new FakeMethods(classToFake, fakedType);
44          collectFakeMethods();
45          registerFakeClassAndItsStates();
46      }
47  
48      private void collectFakeMethods() {
49          Class<?> fakeClass = fake.getClass();
50          new FakeMethodCollector(fakeMethods).collectFakeMethods(fakeClass);
51      }
52  
53      private void registerFakeClassAndItsStates() {
54          fakeMethods.registerFakeStates(fake, forStartupFake);
55  
56          FakeClasses fakeClasses = TestRun.getFakeClasses();
57  
58          if (forStartupFake) {
59              fakeClasses.addFake(fakeMethods.getFakeClassInternalName(), fake);
60          } else {
61              fakeClasses.addFake(fake);
62          }
63      }
64  
65      public void redefineMethods() {
66          @Nullable
67          Class<?> classToModify = realClass;
68  
69          while (classToModify != null && fakeMethods.hasUnusedFakes()) {
70              byte[] modifiedClassFile = modifyRealClass(classToModify);
71  
72              if (modifiedClassFile != null) {
73                  applyClassModifications(classToModify, modifiedClassFile);
74              }
75  
76              Class<?> superClass = classToModify.getSuperclass();
77              classToModify = superClass == Object.class || superClass == Proxy.class ? null : superClass;
78              rcReader = null;
79          }
80      }
81  
82      @Nullable
83      private byte[] modifyRealClass(@Nonnull Class<?> classToModify) {
84          if (rcReader == null) {
85              rcReader = ClassFile.createReaderFromLastRedefinitionIfAny(classToModify);
86          }
87  
88          FakedClassModifier modifier = new FakedClassModifier(rcReader, classToModify, fake, fakeMethods);
89          rcReader.accept(modifier);
90  
91          return modifier.wasModified() ? modifier.toByteArray() : null;
92      }
93  
94      @Nonnull
95      BaseClassModifier createClassModifier(@Nonnull ClassReader cr) {
96          return new FakedClassModifier(cr, realClass, fake, fakeMethods);
97      }
98  
99      void applyClassModifications(@Nonnull Class<?> classToModify, @Nonnull byte[] modifiedClassFile) {
100         ClassDefinition classDef = new ClassDefinition(classToModify, modifiedClassFile);
101         Startup.redefineMethods(classDef);
102 
103         if (forStartupFake) {
104             CachedClassfiles.addClassfile(classToModify, modifiedClassFile);
105         } else {
106             String fakeClassDesc = fakeMethods.getFakeClassInternalName();
107             TestRun.mockFixture().addRedefinedClass(fakeClassDesc, classDef);
108         }
109     }
110 }
