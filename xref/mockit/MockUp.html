5   package mockit;

7   import java.lang.reflect.ParameterizedType;
8   import java.lang.reflect.Type;
9   import java.lang.reflect.TypeVariable;
10  
11  import javax.annotation.Nonnull;
12  
13  import mockit.internal.faking.CaptureOfFakedImplementations;
14  import mockit.internal.faking.FakeClassSetup;
15  import mockit.internal.startup.Startup;
16  
17  /**
18   * A base class used in the creation of a <em>fake</em> for an <em>external</em> type, which is usually a class from
19   * some library or component used from the <em>internal</em> codebase of the system under test (SUT). Such fake classes
20   * can be used as <em>fake implementations</em> for use in unit or integration tests. For example:
21   *
22   * <pre>{@code
23   * public final class FakeSystem <strong>extends MockUp&lt;System></strong> {
24   *    <strong>&#64;Mock</strong> public static long nanoTime() { return 123L; }
25   * }
26   * }</pre>
27   *
28   * One or more <em>fake methods</em> annotated {@linkplain Mock as such} must be defined in the concrete subclass. Each
29   * <code>@Mock</code> method should have a matching method or constructor in the faked class. At runtime, the execution
30   * of a faked method/constructor will get redirected to the corresponding fake method.
31   * <p>
32   * When the type to be faked is specified indirectly through a {@linkplain TypeVariable type variable}, then that type
33   * is taken as a <em>base</em> type whose concrete implementation classes should <em>also</em> get faked. Example:
34   *
35   * <pre>{@code
36   * &#64;Test
37   * public &lt;<strong>BC extends SomeBaseClass</strong>> void someTest() {
38   *     new MockUp&lt;<strong>BC</strong>>() {
39   *        &#64;Mock int someMethod(int i) { return i + 1; }
40   *     };
41   *
42   *     int i = new AConcreteSubclass().someMethod(1);
43   *     assertEquals(2, i);
44   * }
45   * }</pre>
46   *
47   * @param <T>
48   *            specifies the type to be faked; if a type variable is used, then all implementation classes extending or
49   *            implementing that base type are also faked; if the type argument itself is a parameterized type, then only
50   *            its raw type is considered
51   *
52   * @see #MockUp()
53   * @see #MockUp(Class)
54   * @see #onTearDown()
55   * @see #targetType
56   * @see <a href="http://jmockit.github.io/tutorial/Faking.html#setUp" target="tutorial">Tutorial</a>
57   */
58  public class MockUp<T> {
59      static {
60          Startup.verifyInitialization();
61      }
62  
63      /**
64       * Holds the class or generic type targeted by this fake instance.
65       */
66      @Nonnull
67      protected final Type targetType;
68  
69      /**
70       * Applies the {@linkplain Mock fake methods} defined in the concrete subclass to the class specified through the
71       * type parameter.
72       */
73      protected MockUp() {
74          targetType = getTypeToFake();
75          Class<T> classToFake = null;
76  
77          if (targetType instanceof Class<?>) {
78              // noinspection unchecked
79              classToFake = (Class<T>) targetType;
80          } else if (targetType instanceof ParameterizedType) {
81              ParameterizedType parameterizedType = (ParameterizedType) targetType;
82              // noinspection unchecked
83              classToFake = (Class<T>) parameterizedType.getRawType();
84          }
85  
86          if (classToFake != null) {
87              redefineClass(classToFake);
88          } else {
89              Type[] typesToFake = ((TypeVariable<?>) targetType).getBounds();
90  
91              if (typesToFake.length != 1) {
92                  throw new UnsupportedOperationException("Unable to capture more than one base type at once");
93              }
94              new CaptureOfFakedImplementations(this, typesToFake[0]).apply();
95          }
96      }
97  
98      /**
99       * Gets the type to fake.
100      *
101      * @return the type to fake
102      */
103     @Nonnull
104     private Type getTypeToFake() {
105         Class<?> currentClass = getClass();
106 
107         do {
108             Type superclass = currentClass.getGenericSuperclass();
109 
110             if (superclass instanceof ParameterizedType) {
111                 return ((ParameterizedType) superclass).getActualTypeArguments()[0];
112             }
113 
114             if (superclass == MockUp.class) {
115                 throw new IllegalArgumentException("No target type");
116             }
117 
118             currentClass = (Class<?>) superclass;
119         } while (true);
120     }
121 
122     /**
123      * Redefine class.
124      *
125      * @param classToFake
126      *            the class to fake
127      */
128     private void redefineClass(@Nonnull Class<?> classToFake) {
129         if (!classToFake.isInterface()) {
130             new FakeClassSetup(classToFake, this, targetType).redefineMethods();
131         }
132     }
133 
134     /**
135      * Applies the {@linkplain Mock fake methods} defined in the fake class to the given class.
136      * <p>
137      * In most cases, the {@linkplain #MockUp() constructor with no parameters} can be used. This variation is useful
138      * when the type to be faked is not known at compile time. For example, it can be used with an {@linkplain Mock
139      * $advice} method and the <code>fakes</code> system property in order to have an aspect-like fake implementation
140      * applicable to any class; it can then be applied at the beginning of the test run with the desired target class
141      * being specified in the test run configuration.
142      *
143      * @param targetClass
144      *            the target class
145      */
146     protected MockUp(@Nonnull Class<?> targetClass) {
147         targetType = targetClass;
148         redefineClass(targetClass);
149     }
150 
151     /**
152      * An empty method that can be overridden in a fake class that wants to be notified whenever the fake is
153      * automatically torn down. Tear down happens when the fake goes out of scope: at the end of the test when applied
154      * inside a test, at the end of the test class when applied before the test class, or at the end of the test run
155      * when applied through the "<code>fakes</code>" system property.
156      * <p>
157      * By default, this method does nothing.
158      */
159     protected void onTearDown() {
160     }
161 }
